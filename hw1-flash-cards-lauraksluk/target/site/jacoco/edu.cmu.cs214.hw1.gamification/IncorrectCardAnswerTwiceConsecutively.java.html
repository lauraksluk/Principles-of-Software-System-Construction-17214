package edu.cmu.cs214.hw1.gamification;

import edu.cmu.cs214.hw1.cards.CardStatus;

import java.util.List;

public class IncorrectCardAnswerTwiceConsecutively implements Achievement {

    @Override
    public boolean achieved(List<CardStatus> cards) {
        for (CardStatus c : cards) {
            List<Boolean> success = c.getResults();
            int size = success.size();
            if (size > 1) {
                if (!success.get(size - 1) && !success.get(size - 2)) {
                    return true;
                }
            }
        }
        return false;
    }

    @Override
    public String toString() {
        return "You answered this card incorrectly twice in a row!";
    }
}
