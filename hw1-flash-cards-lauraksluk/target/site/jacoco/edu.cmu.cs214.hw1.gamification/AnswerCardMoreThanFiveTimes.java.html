package edu.cmu.cs214.hw1.gamification;

import edu.cmu.cs214.hw1.cards.CardStatus;

import java.util.List;

public class AnswerCardMoreThanFiveTimes implements Achievement {
    private final int threshold = 5;
    /**
     * Determines if the achievement, of answering a card more than 5 times, is met.
     * @param cards The {@link CardStatus} objects
     * @return {@code true} if any card in given list was answered more than 5 times.
     */
    @Override
    public boolean achieved(List<CardStatus> cards) {
        for (CardStatus status : cards) {
            List<Boolean> success = status.getResults();
            if (success.size() >= threshold) {
                return true;
            }
        }
        return false;
    }

    @Override
    public String toString() {
        return "You answered this card more than 5 times!";
    }
}
