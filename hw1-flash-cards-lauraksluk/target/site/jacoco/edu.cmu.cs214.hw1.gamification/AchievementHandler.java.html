package edu.cmu.cs214.hw1.gamification;

import edu.cmu.cs214.hw1.cards.CardStatus;

import java.util.ArrayList;
import java.util.List;

public class AchievementHandler {
    private List<Achievement> all;

    public AchievementHandler() {
        all = new ArrayList<>();
        Achievement fiveTimes = new AnswerCardMoreThanFiveTimes();
        Achievement allCorrect = new AllCorrectLastRound();
        Achievement incorrectTwice = new IncorrectCardAnswerTwiceConsecutively();
        all.add(fiveTimes);
        all.add(allCorrect);
        all.add(incorrectTwice);
    }

    public List<Achievement> getAll() {
        return all;
    }

    public List<String> getNewAchievements(List<CardStatus> cards) {
        List<String> result = new ArrayList<>();
        List<Achievement> toRemove = new ArrayList<>();
        for (Achievement a : all) {
            if (a.achieved(cards)) {
                result.add(a.toString());
                toRemove.add(a);
            }
        }
        for (Achievement a : toRemove) {
            all.remove(a);
        }
        return result;
    }

}
