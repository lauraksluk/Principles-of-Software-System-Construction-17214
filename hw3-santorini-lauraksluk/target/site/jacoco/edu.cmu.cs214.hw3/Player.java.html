package edu.cmu.cs214.hw3;

import java.util.ArrayList;
import java.util.List;

/**
 * Player class that represents each player in the Santorini game.
 * Each player has 2 workers, represented by a list of Workers.
 */
public class Player {
    private int playerId;
    private int numWorkers;
    private List<Worker> workers;

    /**
     * Constructor for creating a new player.
     * @param id player id
     */
    public Player(int id) {
        playerId = id;
        numWorkers = 2;
        workers = new ArrayList<>(numWorkers);
        initializeWorkers(playerId);
    }

    /**
     * Method (private) that initializes the list of workers with
     * 2 workers that have the same id as the player.
     * @param playerId player id
     */
    private void initializeWorkers(int playerId) {
        for (int i = 0; i < numWorkers; i++) {
            workers.add(new Worker(playerId));
        }
    }

    /**
     * Method that returns the Workers list.
     * @return list of Workers that belong to this player.
     */
    public List<Worker> getWorkers() {
        return workers;
    }

    /**
     * Method that returns the player id.
     * @return player id
     */
    public int getPlayerId() {
        return playerId;
    }

    /**
     * Method that returns the number of workers the player has.
     * @return number of workers belonging to player.
     */
    public int getNumWorkers() {
        return numWorkers;
    }

}
